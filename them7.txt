/* ── Mouse-zone career image slider ── */
(function(){
  const slides   = document.querySelectorAll('.img-slide');
  const thumbs   = document.querySelectorAll('.img-thumb');
  const label    = document.getElementById('imgLabel');
  const num      = document.getElementById('imgNum');
  const progress = document.getElementById('imgProgress');
  const cursor   = document.getElementById('careerCursor');
  const right    = document.getElementById('careerRight');
  const zones    = document.querySelectorAll('.img-zone');
  let current = 0;
  let autoTimer, progressTimer, progressVal = 0;

  function activate(i){
    if(i === current) return;
    slides[current].classList.remove('active');
    thumbs[current].classList.remove('active');
    current = i;
    slides[current].classList.add('active');
    thumbs[current].classList.add('active');
    label.textContent = slides[current].dataset.label;
    num.textContent   = (current+1) + ' / ' + slides.length;
    resetProgress();
  }

  function resetProgress(){
    progressVal = 0;
    progress.style.transition = 'none';
    progress.style.width = '0%';
    clearInterval(progressTimer);
    setTimeout(() => {
      progress.style.transition = 'width 4s linear';
      progress.style.width = '100%';
    }, 30);
  }

  function startAuto(){
    clearInterval(autoTimer);
    autoTimer = setInterval(()=>{ activate((current+1)%slides.length); }, 4500);
  }

  // Mouse-zone switching
  zones.forEach(z => {
    z.addEventListener('mouseenter', () => {
      activate(parseInt(z.dataset.idx));
      clearInterval(autoTimer);
    });
    z.addEventListener('mouseleave', () => { startAuto(); });
  });

  // Thumbnail click
  thumbs.forEach(t => {
    t.addEventListener('click', () => {
      activate(parseInt(t.dataset.idx));
      startAuto();
    });
  });

  // Custom cursor
  if(right){
    right.addEventListener('mousemove', e => {
      const r = right.getBoundingClientRect();
      cursor.style.left = (e.clientX - r.left) + 'px';
      cursor.style.top  = (e.clientY - r.top)  + 'px';
      cursor.style.opacity = '1';
    });
    right.addEventListener('mouseleave', () => { cursor.style.opacity='0'; startAuto(); });
    right.addEventListener('mouseenter', () => { cursor.style.opacity='1'; });
  }

  resetProgress();
  startAuto();
})();

/* ── SIP calculator ── */
function calcSip(){
  const P = parseFloat(document.getElementById('sipAmt').value)||0;
  const n = parseInt(document.getElementById('sipFreq').value)||12;
  const Y = parseFloat(document.getElementById('sipYrs').value)||0;
  let R = parseFloat(document.getElementById('sipRate').value)||0;
  R = Math.max(0, R - (parseFloat(document.getElementById('sipInfl').value)||0));
  const r = R/100/n, tt = n*Y;
  const mat = r>0 ? P*((Math.pow(1+r,tt)-1)/r)*(1+r) : P*tt;
  const inv = P*tt;
  const fmt = v => '₹'+Math.round(v).toLocaleString('en-IN');
  document.getElementById('sInv').textContent   = fmt(inv);
  document.getElementById('sGain').textContent  = fmt(mat-inv);
  document.getElementById('sTotal').textContent = fmt(mat);
  document.getElementById('sipRes').classList.add('show');
}
